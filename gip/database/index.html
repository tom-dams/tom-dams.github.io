Database Mail

create schema gip_taak_08_DamsTom;
use gip_taak_08_DamsTom;

create table UserAccount(
    id int not null auto_increment primary key,
    wachtwoord text not null,
    email text not null 
);

create table Mail(
    id int not null auto_increment primary key,
    user_id int not null,
    tijdstip datetime,
    titel text not null,
    inhoud text not null,
    bijlage text,
    ontvanger text not null 
);

insert into UserAccount(id, wachtwoord, email)
values
(1, "Test123", "test@gmail.com"),
(2, "Test456", "test2@gmail.com");

insert into Mail(mail_id, user_id, tijdstip, titel, inhoud, bijlage, ontvanger)
values
(1,1, SYSDATE(), "Titel", "Dit is de inhoud", "Bijlage", "test2@gmail.com"),
(2, 2, sysdate(), "Titel2", "Hoi", "", "test@gmail.com"); 

select * from UserAccount join Mail on UserAccount.id = Mail.user_id;
